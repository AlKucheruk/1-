Функция НормализоватьЦифровойКод(ЗначениеЯчейки)
    Если НЕ ЗначениеЗаполнено(ЗначениеЯчейки) Тогда
        Возврат "";
    КонецЕсли;

    Текст = СокрЛП(Строка(ЗначениеЯчейки));
    Текст = СтрЗаменить(Текст, Символ(160), "");
    Текст = СтрЗаменить(Текст, " ", "");

    Если СтрНайти(Текст, ".") > 0 Тогда
        Части = СтрРазделить(Текст, ".");
        Если Части.Количество() = 2 Тогда
            ДробнаяЧасть = Части[1];
            ЕстьТолькоНули = Истина;
            Для Сч = 1 По СтрДлина(ДробнаяЧасть) Цикл
                Если Сред(ДробнаяЧасть, Сч, 1) <> "0" Тогда
                    ЕстьТолькоНули = Ложь;
                    Прервать;
                КонецЕсли;
            КонецЦикла;
            Если ЕстьТолькоНули Тогда
                Текст = Части[0];
            КонецЕсли;
        КонецЕсли;
    КонецЕсли;

    Если СтрНайти(ВРег(Текст), "E") > 0 Тогда
        Попытка
            ЧисловоеЗначение = Число(СтрЗаменить(Текст, "E", "e"));
            Текст = Формат(ЧисловоеЗначение, "ЧДЦ=0; ЧРД=.; ЧГ=0");
            Если СтрНайти(Текст, ".") > 0 Тогда
                Текст = СтрРазделить(Текст, ".")[0];
            КонецЕсли;
        Исключение
        КонецПопытки;
    КонецЕсли;

    Результат = "";
    Для Сч = 1 По СтрДлина(Текст) Цикл
        Симв = Сред(Текст, Сч, 1);
        Если Найти("0123456789", Симв) > 0 Тогда
            Результат = Результат + Симв;
        КонецЕсли;
    КонецЦикла;

    Возврат Результат;
КонецФункции

Функция ПарситьДату(ТекстДата)
    Текст = СокрЛП(ТекстДата);
    Если ПустаяСтрока(Текст) Тогда
        Возврат '00010101';
    КонецЕсли;

    Если СтрДлина(Текст) >= 10 И Сред(Текст, 3, 1) = "." И Сред(Текст, 6, 1) = "." Тогда
        День  = Лев(Текст, 2);
        Месяц = Сред(Текст, 4, 2);
        Год   = Сред(Текст, 7, 4);
        Попытка
            Возврат Дата(Год + Месяц + День);
        Исключение
        КонецПопытки;
    КонецЕсли;

    Если СтрДлина(Текст) >= 10 И Сред(Текст, 5, 1) = "-" И Сред(Текст, 8, 1) = "-" Тогда
        Год   = Лев(Текст, 4);
        Месяц = Сред(Текст, 6, 2);
        День  = Сред(Текст, 9, 2);
        Попытка
            Возврат Дата(Год + Месяц + День);
        Исключение
        КонецПопытки;
    КонецЕсли;

    Если СтрДлина(Текст) = 8 Тогда
        Попытка
            Возврат Дата(Текст);
        Исключение
        КонецПопытки;
    КонецЕсли;

    Возврат '00010101';
КонецФункции

                    НовСтр.ЭтоДубль = Ложь;
                КонецЕсли;
            КонецЕсли;
        Исключение
            НовСтр.ЭтоДубль = Ложь;
            НовСтр.Комментарий = НовСтр.Комментарий + "Ошибка проверки дубля. ";
            LogException("Проверка дубля, строка " + Строка(НомСтр));
        КонецПопытки;

        // --- Определение статуса ---
        Если НовСтр.ЭтоДубль Тогда
            НовСтр.Статус = "Дубль";
        ИначеЕсли НовСтр.НайденКонтрагент
            И НовСтр.НайденаСтатья
            И НовСтр.НайденаОрганизация
            И НовСтр.НайденСчетОрганизации
            И ЗначениеЗаполнено(НовСтр.Дата)
            И НовСтр.Дата <> '00010101'
            И НовСтр.Сумма > 0
            И НЕ ПустаяСтрока(НовСтр.УИД) Тогда

            НовСтр.Статус = "Готов";
        Иначе
            НовСтр.Статус = "Ошибка";
            КоличествоОшибок = КоличествоОшибок + 1;
        КонецЕсли;

// Логируем только первые 10 строк данных
        Если СчётчикСтрок <= 10 Тогда
            LogLine("Стр." + Строка(НовСтр.НомерСтроки)
                + " | " + НовСтр.Статус
                + " | УИД=" + НовСтр.УИД
                + " | Контр=" + НовСтр.Контрагент
                + " | ОперацияУТ=" + НовСтр.ОперацияУТ
                + " | Сумма=" + Строка(НовСтр.Сумма),
                СчётчикСтрок);
        КонецЕсли;

        КоличествоЗагружено = КоличествоЗагружено + 1;

    КонецЦикла;

    LogLine("=== ИТОГ: Загружено=" + Строка(КоличествоЗагружено)
        + ", Пропущено=" + Строка(КоличествоПропущено)
        + ", Ошибок=" + Строка(КоличествоОшибок)
        + ", Дублей=" + Строка(КоличествоДублей) + " ===");

    Возврат ТЗ;

    Если СтрНайти(НормНапр, "ПРИХОД") > 0

// *** ИЗМЕНЕНО v09: используем ОперацияУТ (колонка 31) вместо ВидОперации (колонка 13) ***
&НаСервереБезКонтекста
Функция ПолучитьХозяйственнуюОперацию(ЭтоПоступление, ОперацияУТ)

    НормВид = ВРег(СокрЛП(ОперацияУТ));

    Если ЭтоПоступление Тогда
        Если СтрНайти(НормВид, "ОПЛАТА ОТ КЛИЕНТА") > 0
            Или СтрНайти(НормВид, "ПОСТУПЛЕНИЕ ОПЛАТЫ ОТ КЛИЕНТА") > 0 Тогда
            Возврат Перечисления.ХозяйственныеОперации.ОплатаОтКлиента;
        ИначеЕсли СтрНайти(НормВид, "ВОЗВРАТ") > 0 И СтрНайти(НормВид, "ПОСТАВЩИК") > 0 Тогда
            Возврат Перечисления.ХозяйственныеОперации.ВозвратДенежныхСредствОтПоставщика;
        ИначеЕсли СтрНайти(НормВид, "ИНКАССАЦИЯ") > 0 Тогда
            Возврат Перечисления.ХозяйственныеОперации.ПоступлениеДенежныхСредствИзКассыККМ;
        ИначеЕсли СтрНайти(НормВид, "ДРУГОЙ СЧЕТ") > 0
            Или СтрНайти(НормВид, "ДРУГОГО СЧЕТ") > 0 Тогда
            Возврат Перечисления.ХозяйственныеОперации.ПоступлениеДенежныхСредствСДругогоСчета;
        Иначе
            Возврат Перечисления.ХозяйственныеОперации.ПрочееПоступлениеДенежныхСредств;
        КонецЕсли;
    Иначе
        Если СтрНайти(НормВид, "ОПЛАТА ПОСТАВЩИКУ") > 0 Тогда
            Возврат Перечисления.ХозяйственныеОперации.ОплатаПоставщику;
        ИначеЕсли СтрНайти(НормВид, "ВОЗВРАТ") > 0 И СтрНайти(НормВид, "КЛИЕНТ") > 0 Тогда
            Возврат Перечисления.ХозяйственныеОперации.ВозвратДенежныхСредствКлиенту;
        ИначеЕсли СтрНайти(НормВид, "НАЛОГ") > 0
            Или СтрНайти(НормВид, "ВЗНОС") > 0
            Или СтрНайти(НормВид, "ПЕРЕЧИСЛЕНИЕ НАЛОГ") > 0 Тогда
            Возврат Перечисления.ХозяйственныеОперации.ПеречислениеНалоговВзнос;
        ИначеЕсли СтрНайти(НормВид, "ЗАРПЛАТ") > 0
            Или СтрНайти(НормВид, "ВЕДОМОСТ") > 0 Тогда
            Возврат Перечисления.ХозяйственныеОперации.ВыплатаЗаработнойПлаты;
        ИначеЕсли СтрНайти(НормВид, "ПОДОТЧЕТ") > 0 Тогда
            Возврат Перечисления.ХозяйственныеОперации.ВыдачаПодотчетнику;
        ИначеЕсли СтрНайти(НормВид, "ДРУГОЙ СЧЕТ") > 0
            Или СтрНайти(НормВид, "ДРУГОГО СЧЕТ") > 0 Тогда
            Возврат Перечисления.ХозяйственныеОперации.ПеречислениеДенежныхСредствНаДругойСчет;
        Иначе
            Возврат Перечисления.ХозяйственныеОперации.ПрочееСписаниеДенежныхСредств;
        КонецЕсли;
    КонецЕсли;

Функция ЗагрузитьВБазуСервер() Экспорт

    Если ТаблицаДанных.Количество() = 0 Тогда
        Возврат "Таблица данных пуста. Сначала выберите и загрузите файл.";
    КонецЕсли;

    КоличествоГотово     = 0;
    КоличествоЗагружено  = 0;
    КоличествоДублей     = 0;
    КоличествоОшибок     = 0;
    КоличествоПропущено  = 0;

    ДатаЗагрузки = ТекущаяДата();
    ЛокИмяФайла  = СокрЛП(ИмяФайла);

    Для Каждого Стр Из ТаблицаДанных Цикл

        Если Стр.Статус <> "Готов" Тогда
            КоличествоПропущено = КоличествоПропущено + 1;
            Продолжить;
        КонецЕсли;

        КоличествоГотово = КоличествоГотово + 1;

        НачатьТранзакцию();
        Попытка

            Если НайтиДубльПоУИД(Стр.УИД) <> Неопределено Тогда
                ЗафиксироватьТранзакцию();
                Стр.Статус     = "Дубль";
                Стр.Комментарий = Стр.Комментарий + " Повторная проверка: уже загружен.";
                КоличествоДублей = КоличествоДублей + 1;
                Продолжить;
            КонецЕсли;

            СсылкаДокумента = СоздатьДокументПоСтроке(Стр, ЛокИмяФайла, ДатаЗагрузки);

            ЗафиксироватьТранзакцию();

            Стр.Статус      = "Загружен";
            Стр.Комментарий = "Создан документ: " + Строка(СсылкаДокумента);
            КоличествоЗагружено = КоличествоЗагружено + 1;

            LogLine("ЗАГРУЖЕНО | Стр." + Строка(Стр.НомерСтроки)
                + " | УИД=" + Стр.УИД
                + " | " + Строка(СсылкаДокумента));

        Исключение
            Попытка
                Если ТранзакцияАктивна() Тогда
                    ОтменитьТранзакцию();
                КонецЕсли;
            Исключение
            КонецПопытки;

            Стр.Статус      = "Ошибка";
            Стр.Комментарий = "Ошибка загрузки: " + ОписаниеОшибки();
            КоличествоОшибок = КоличествоОшибок + 1;

            LogException("ЗагрузитьВБазуСервер, строка "
                + Строка(Стр.НомерСтроки) + ", УИД=" + Стр.УИД);
        КонецПопытки;

    КонецЦикла;

    Если КоличествоГотово = 0 Тогда
        Возврат "Нет строк со статусом ""Готов"" для загрузки.";
    КонецЕсли;

    Возврат "Загрузка завершена. Готово: " + Строка(КоличествоГотово)
        + "; Загружено: " + Строка(КоличествоЗагружено)
        + "; Дубли: " + Строка(КоличествоДублей)
        + "; Ошибки: " + Строка(КоличествоОшибок)
        + "; Пропущено: " + Строка(КоличествоПропущено) + ".";
КонецФункции
