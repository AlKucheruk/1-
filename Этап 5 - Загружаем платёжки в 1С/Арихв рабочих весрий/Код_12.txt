Процедура LogLine(Текст, СчётчикСтрок = 0)
    // Логируем только если это не построчный лог (СчётчикСтрок = 0)
    // или первые 10 строк данных
    Если СчётчикСтрок > 10 Тогда
        Возврат;
    КонецЕсли;

    Попытка
        ПутьКаталога = "D:\1с\КАА";
        Попытка
            СоздатьКаталог(ПутьКаталога);
        Исключение
        КонецПопытки;

        Если Прав(ПутьКаталога, 1) <> "\" Тогда
            ПутьКаталога = ПутьКаталога + "\";
        КонецЕсли;

        ПутьФайла = ПутьКаталога + "zagr_bp_debug.log";
        СтрокаЛога = Формат(ТекущаяДата(), "ДЛФ=dd.MM.yyyy ЧЧ:мм:сс") + " | " + Текст;

        ТД = Новый ТекстовыйДокумент;
        Попытка
            ТД.Прочитать(ПутьФайла, КодировкаТекста.UTF8);
        Исключение
            ТД = Новый ТекстовыйДокумент;
        КонецПопытки;

        ТД.ДобавитьСтроку(СтрокаЛога);
        ТД.Записать(ПутьФайла, КодировкаТекста.UTF8);
    Исключение
    КонецПопытки;
КонецПроцедуры

    Исключение
    КонецПопытки;
КонецПроцедуры

Функция НормализоватьЦифровойКод(ЗначениеЯчейки)
    Если НЕ ЗначениеЗаполнено(ЗначениеЯчейки) Тогда
        Возврат "";
    КонецЕсли;

    Текст = СокрЛП(Строка(ЗначениеЯчейки));
    Текст = СтрЗаменить(Текст, Символ(160), "");
    Текст = СтрЗаменить(Текст, " ", "");

    Если СтрНайти(Текст, ".") > 0 Тогда
        Части = СтрРазделить(Текст, ".");
        Если Части.Количество() = 2 Тогда
            ДробнаяЧасть = Части[1];
            ЕстьТолькоНули = Истина;
            Для Сч = 1 По СтрДлина(ДробнаяЧасть) Цикл
                Если Сред(ДробнаяЧасть, Сч, 1) <> "0" Тогда
                    ЕстьТолькоНули = Ложь;
                    Прервать;
                КонецЕсли;
            КонецЦикла;
            Если ЕстьТолькоНули Тогда
                Текст = Части[0];
            КонецЕсли;
        КонецЕсли;
    КонецЕсли;

    Если СтрНайти(ВРег(Текст), "E") > 0 Тогда
        Попытка
            ЧисловоеЗначение = Число(СтрЗаменить(Текст, "E", "e"));
            Текст = Формат(ЧисловоеЗначение, "ЧДЦ=0; ЧРД=.; ЧГ=0");
            Если СтрНайти(Текст, ".") > 0 Тогда
                Текст = СтрРазделить(Текст, ".")[0];
            КонецЕсли;
        Исключение
        КонецПопытки;
    КонецЕсли;

    Результат = "";
    Для Сч = 1 По СтрДлина(Текст) Цикл
        Симв = Сред(Текст, Сч, 1);
        Если Найти("0123456789", Симв) > 0 Тогда
            Результат = Результат + Симв;
        КонецЕсли;
    КонецЦикла;

    Возврат Результат;
КонецФункции

Функция ПарситьДату(ТекстДата)
    Текст = СокрЛП(ТекстДата);
    Если ПустаяСтрока(Текст) Тогда
        Возврат '00010101';
    КонецЕсли;

    Если СтрДлина(Текст) >= 10 И Сред(Текст, 3, 1) = "." И Сред(Текст, 6, 1) = "." Тогда
        День  = Лев(Текст, 2);
        Месяц = Сред(Текст, 4, 2);
        Год   = Сред(Текст, 7, 4);
        Попытка
            Возврат Дата(Год + Месяц + День);
        Исключение
        КонецПопытки;
    КонецЕсли;

    Если СтрДлина(Текст) >= 10 И Сред(Текст, 5, 1) = "-" И Сред(Текст, 8, 1) = "-" Тогда
        Год   = Лев(Текст, 4);
        Месяц = Сред(Текст, 6, 2);
        День  = Сред(Текст, 9, 2);
        Попытка
            Возврат Дата(Год + Месяц + День);
        Исключение
        КонецПопытки;
    КонецЕсли;

    Если СтрДлина(Текст) = 8 Тогда
        Попытка
            Возврат Дата(Текст);
        Исключение
        КонецПопытки;
    КонецЕсли;

    Возврат '00010101';
КонецФункции

                НовСтр.БанковскийСчетОрганизации = НайтиБанковскийСчетОрганизации(НовСтр.ОрганизацияСсылка, НовСтр.НашСчет);
                НовСтр.НайденСчетОрганизации = ЗначениеЗаполнено(НовСтр.БанковскийСчетОрганизации);
                Если НЕ НовСтр.НайденСчетОрганизации Тогда
                    НовСтр.Комментарий = НовСтр.Комментарий + "Банк. счёт организации не найден. ";
                КонецЕсли;
            Исключение
                НовСтр.НайденСчетОрганизации = Ложь;
                НовСтр.Комментарий = НовСтр.Комментарий + "Ошибка поиска банк. счёта организации. ";
                LogException("Поиск банк. счёта организации, строка " + Строка(НомСтр));
            КонецПопытки;
        КонецЕсли;

        // --- Поиск банковского счёта контрагента ---
        Если НовСтр.НайденКонтрагент И НЕ ПустаяСтрока(НовСтр.СчетКонтр) Тогда
            Попытка
                НовСтр.БанковскийСчетКонтрагента = НайтиБанковскийСчетКонтрагента(НовСтр.КонтрагентСсылка, НовСтр.СчетКонтр);
                НовСтр.НайденСчетКонтрагента = ЗначениеЗаполнено(НовСтр.БанковскийСчетКонтрагента);
            Исключение
                НовСтр.НайденСчетКонтрагента = Ложь;
                LogException("Поиск банк. счёта контрагента, строка " + Строка(НомСтр));
            КонецПопытки;
        КонецЕсли;

        // --- Поиск статьи ДДС ---
        Попытка
            НовСтр.СтатьяДДССсылка = НайтиСтатьюДДС(НовСтр.СтатьяДДС);
            НовСтр.НайденаСтатья = ЗначениеЗаполнено(НовСтр.СтатьяДДССсылка);
            Если НЕ НовСтр.НайденаСтатья Тогда
                НовСтр.Комментарий = НовСтр.Комментарий + "Статья ДДС не найдена. ";
            КонецЕсли;
        Исключение
            НовСтр.НайденаСтатья = Ложь;
            НовСтр.Комментарий = НовСтр.Комментарий + "Ошибка поиска статьи ДДС. ";
            LogException("Поиск статьи ДДС, строка " + Строка(НомСтр));
        КонецПопытки;

        // --- Проверка дубля по УИД ---
        Попытка
            Если ПустаяСтрока(НовСтр.УИД) Тогда
                НовСтр.ЭтоДубль = Ложь;
                НовСтр.Комментарий = НовСтр.Комментарий + "УИД пустой. ";
            Иначе
                СвязанныйДок = НайтиДубльПоУИД(НовСтр.УИД);
                Если СвязанныйДок <> Неопределено Тогда
                    НовСтр.ЭтоДубль = Истина;
                    НовСтр.КомментарийДубля = "Загружен " + Формат(СвязанныйДок.ДатаЗагрузки, "ДЛФ=DD") + " из " + СвязанныйДок.ИмяФайла;
                    КоличествоДублей = КоличествоДублей + 1;
                Иначе
                    НовСтр.ЭтоДубль = Ложь;
                КонецЕсли;
            КонецЕсли;
        Исключение
            НовСтр.ЭтоДубль = Ложь;
            НовСтр.Комментарий = НовСтр.Комментарий + "Ошибка проверки дубля. ";
            LogException("Проверка дубля, строка " + Строка(НомСтр));
        КонецПопытки;

        // --- Определение статуса ---
        Если НовСтр.ЭтоДубль Тогда
            НовСтр.Статус = "Дубль";
        ИначеЕсли НовСтр.НайденКонтрагент
            И НовСтр.НайденаСтатья
            И НовСтр.НайденаОрганизация
            И НовСтр.НайденСчетОрганизации
            И ЗначениеЗаполнено(НовСтр.Дата)
            И НовСтр.Дата <> '00010101'
            И НовСтр.Сумма > 0
            И НЕ ПустаяСтрока(НовСтр.УИД) Тогда

            НовСтр.Статус = "Готов";
        Иначе
            НовСтр.Статус = "Ошибка";
            КоличествоОшибок = КоличествоОшибок + 1;
        КонецЕсли;

        // Логируем только первые 10 строк данных
        Если СчётчикСтрок <= 10 Тогда
            LogLine("Стр." + Строка(НовСтр.НомерСтроки)
                + " | " + НовСтр.Статус
                + " | " + НовСтр.Контрагент
                + " | " + Формат(НовСтр.Сумма, "ЧДЦ=2")
                + " | " + НовСтр.Комментарий, СчётчикСтрок);
        ИначеЕсли СчётчикСтрок = 11 Тогда
            LogLine("... далее построчный лог отключён ...");
        КонецЕсли;

        КоличествоЗагружено = КоличествоЗагружено + 1;

    КонецЦикла;

    LogLine("=== ИТОГ: Загружено=" + Строка(КоличествоЗагружено)
        + " Пропущено=" + Строка(КоличествоПропущено)
        + " Ошибок=" + Строка(КоличествоОшибок)
        + " Дублей=" + Строка(КоличествоДублей) + " ===");

    Возврат ТЗ;

КонецФункции

// Функция ищет хозяйственную операцию по строковому наименованию.
// Значение перечисления ХозяйственныеОперации определяется по наименованию из Excel.
//
// Параметры:
//   НаименованиеОперации - Строка - текстовое наименование операции из колонки "Операция УТ"
//
// Возвращаемое значение:
//   Перечисления.ХозяйственныеОперации - найденное значение перечисления
//   Неопределено - если операция не найдена
//
&НаСервереБезКонтекста
Функция ОпределитьХозяйственнуюОперацию(НаименованиеОперации)
    
    Если ПустаяСтрока(НаименованиеОперации) Тогда
        Возврат Неопределено;
    КонецЕсли;
    
    НаимВРег = ВРег(СокрЛП(НаименованиеОперации));
    
    // Соответствие текстовых наименований операций из Excel-файла (колонка "Операция УТ")
    // реальным значениям перечисления Перечисления.ХозяйственныеОперации из базы УТ 11.5
    // Имена перечислений взяты из выгрузки ХозяйственныеОперации.txt (реальные данные базы)
    СоответствиеОпераций = Новый Соответствие;
    
    // ===== ПОСТУПЛЕНИЯ (ПоступлениеБезналичныхДенежныхСредств) =====
    
    СоответствиеОпераций.Вставить("ПОСТУПЛЕНИЕ ОПЛАТЫ ОТ КЛИЕНТА",
        Перечисления.ХозяйственныеОперации.ПоступлениеОплатыОтКлиента);
    
    // Возврат от поставщика (имя: ВозвратДенежныхСредствОтПоставщика, синоним: "Возврат или иное поступление от поставщика")
    СоответствиеОпераций.Вставить("ВОЗВРАТ ИЛИ ИНОЕ ПОСТУПЛЕНИЕ ОТ ПОСТАВЩИКА",
        Перечисления.ХозяйственныеОперации.ВозвратДенежныхСредствОтПоставщика);
    СоответствиеОпераций.Вставить("ВОЗВРАТ ОПЛАТЫ ОТ ПОСТАВЩИКА",
        Перечисления.ХозяйственныеОперации.ВозвратДенежныхСредствОтПоставщика);
    СоответствиеОпераций.Вставить("ВОЗВРАТ ОТ ПОСТАВЩИКА",
        Перечисления.ХозяйственныеОперации.ВозвратДенежныхСредствОтПоставщика);
    
    // Прочее поступление (имя: ПрочееПоступлениеДенежныхСредств, синоним: "Прочее поступление ДС")
    СоответствиеОпераций.Вставить("ПРОЧЕЕ ПОСТУПЛЕНИЕ ДС",
        Перечисления.ХозяйственныеОперации.ПрочееПоступлениеДенежныхСредств);
    СоответствиеОпераций.Вставить("ПРОЧЕЕ ПОСТУПЛЕНИЕ ДЕНЕЖНЫХ СРЕДСТВ",
        Перечисления.ХозяйственныеОперации.ПрочееПоступлениеДенежныхСредств);
    СоответствиеОпераций.Вставить("ПРОЧЕЕ ПОСТУПЛЕНИЕ БЕЗНАЛИЧНЫХ ДЕНЕЖНЫХ СРЕДСТВ",
        Перечисления.ХозяйственныеОперации.ПрочееПоступлениеДенежныхСредств);
    СоответствиеОпераций.Вставить("ПРОЧЕЕ ПОСТУПЛЕНИЕ",
        Перечисления.ХозяйственныеОперации.ПрочееПоступлениеДенежныхСредств);
    
    // Конвертация валюты (имя: ПоступлениеДенежныхСредствОтКонвертацииВалюты, синоним: "Поступление ДС от конвертации валюты")
    СоответствиеОпераций.Вставить("ПОСТУПЛЕНИЕ ДС ОТ КОНВЕРТАЦИИ ВАЛЮТЫ",
        Перечисления.ХозяйственныеОперации.ПоступлениеДенежныхСредствОтКонвертацииВалюты);
    СоответствиеОпераций.Вставить("ПОСТУПЛЕНИЕ ОТ ПРОДАЖИ ИНОСТРАННОЙ ВАЛЮТЫ",
        Перечисления.ХозяйственныеОперации.ПоступлениеДенежныхСредствОтКонвертацииВалюты);
    
    // Инкассация (имя: ИнкассацияДенежныхСредствИзБанка, синоним: "Инкассация ДС из банка")
    СоответствиеОпераций.Вставить("ИНКАССАЦИЯ",
        Перечисления.ХозяйственныеОперации.ИнкассацияДенежныхСредствИзБанка);
    СоответствиеОпераций.Вставить("ИНКАССАЦИЯ ДС ИЗ БАНКА",
        Перечисления.ХозяйственныеОперации.ИнкассацияДенежныхСредствИзБанка);
    
    // Возврат от подотчетника (имя: ВозвратДенежныхСредствОтПодотчетника, синоним: "Возврат ДС от подотчетника")
    СоответствиеОпераций.Вставить("ВОЗВРАТ ДС ОТ ПОДОТЧЕТНИКА",
        Перечисления.ХозяйственныеОперации.ВозвратДенежныхСредствОтПодотчетника);
    СоответствиеОпераций.Вставить("ВОЗВРАТ ДЕНЕЖНЫХ СРЕДСТВ ОТ ПОДОТЧЕТНИКА",
        Перечисления.ХозяйственныеОперации.ВозвратДенежныхСредствОтПодотчетника);
    
    // Поступление от другой организации
    СоответствиеОпераций.Вставить("ПОСТУПЛЕНИЕ ДС ОТ ДРУГОЙ ОРГАНИЗАЦИИ",
        Перечисления.ХозяйственныеОперации.ПоступлениеДенежныхСредствИзДругойОрганизации);
    СоответствиеОпераций.Вставить("ПОСТУПЛЕНИЕ ДЕНЕЖНЫХ СРЕДСТВ ИЗ ДРУГОЙ ОРГАНИЗАЦИИ",
        Перечисления.ХозяйственныеОперации.ПоступлениеДенежныхСредствИзДругойОрганизации);
    
    // Поступление из другой кассы
    СоответствиеОпераций.Вставить("ПОСТУПЛЕНИЕ ДС ИЗ ДРУГОЙ КАССЫ",
        Перечисления.ХозяйственныеОперации.ПоступлениеДенежныхСредствИзДругойКассы);
    СоответствиеОпераций.Вставить("ПОСТУПЛЕНИЕ ДЕНЕЖНЫХ СРЕДСТВ ИЗ ДРУГОЙ КАССЫ",
        Перечисления.ХозяйственныеОперации.ПоступлениеДенежныхСредствИзДругойКассы);
    
    // Кредиты и займы полученные (имя: ПоступлениеДенежныхСредствПоКредитам, синоним: "Поступление по кредитам и займам полученным")
    СоответствиеОпераций.Вставить("ПОСТУПЛЕНИЕ ПО КРЕДИТАМ И ЗАЙМАМ ПОЛУЧЕННЫМ",
        Перечисления.ХозяйственныеОперации.ПоступлениеДенежныхСредствПоКредитам);
    СоответствиеОпераций.Вставить("ПОЛУЧЕНИЕ КРЕДИТА В БАНКЕ",
        Перечисления.ХозяйственныеОперации.ПоступлениеДенежныхСредствПоКредитам);
    СоответствиеОпераций.Вставить("ПОЛУЧЕНИЕ ЗАЙМА ОТ КОНТРАГЕНТА",
        Перечисления.ХозяйственныеОперации.ПоступлениеДенежныхСредствПоКредитам);
    
    // Возврат налогов
    СоответствиеОпераций.Вставить("ВОЗВРАТ НАЛОГОВ",
        Перечисления.ХозяйственныеОперации.ВозвратНалогов);
    
    // Погашение займа контрагентом (имя: ПоступлениеДенежныхСредствПоЗаймамВыданным)
    СоответствиеОпераций.Вставить("ПОГАШЕНИЕ ЗАЙМА КОНТРАГЕНТОМ",
        Перечисления.ХозяйственныеОперации.ПоступлениеДенежныхСредствПоЗаймамВыданным);
    
    // Внутреннее поступление ДС
    СоответствиеОпераций.Вставить("ВНУТРЕННЕЕ ПОСТУПЛЕНИЕ ДС",

// Функция ищет подходящий договор контрагента с нужной целью.
// Для Поступлений — ищем договор с целью "С покупателем" (Реализация).
// Для Списаний — ищем договор с целью "С поставщиком" (Закупка).
// Если договоров несколько — берём самый ранний по дате создания.
// Если договора нет — создаём новый.
//
// Параметры:
//   КонтрагентСсылка  - СправочникСсылка.Контрагенты - контрагент
//   ОрганизацияСсылка - СправочникСсылка.Организации - организация
//   ЭтоПоступление    - Булево - Истина для поступления, Ложь для списания
//   НаименованиеДоговораИзExcel - Строка - наименование договора из файла (для поиска)
//
// Возвращаемое значение:
//   СправочникСсылка.ДоговорыКонтрагентов - найденный или созданный договор
//
&НаСервереБезКонтекста
Функция ПолучитьИлиСоздатьДоговор(КонтрагентСсылка, ОрганизацияСсылка, ЭтоПоступление, НаименованиеДоговораИзExcel = "")
    
    Если НЕ ЗначениеЗаполнено(КонтрагентСсылка) Тогда
        Возврат Справочники.ДоговорыКонтрагентов.ПустаяСсылка();
    КонецЕсли;
    
    // Определяем цель договора (тип) в зависимости от направления
    // В УТ 11.5 ТипДоговора — это перечисление ТипыДоговоров
    Если ЭтоПоступление Тогда
        ЦельДоговора = Перечисления.ТипыДоговоров.СПокупателем;
    Иначе
        ЦельДоговора = Перечисления.ТипыДоговоров.СПоставщиком;
    КонецЕсли;
    
    // 1. Сначала ищем по наименованию из Excel (если указано)
    Если НЕ ПустаяСтрока(НаименованиеДоговораИзExcel) Тогда
        Попытка
            Запрос = Новый Запрос;
            Запрос.Текст =
            "ВЫБРАТЬ ПЕРВЫЕ 1
            |   ДоговорыКонтрагентов.Ссылка КАК Ссылка
            |ИЗ
            |   Справочник.ДоговорыКонтрагентов КАК ДоговорыКонтрагентов
            |ГДЕ
            |   ДоговорыКонтрагентов.Контрагент = &Контрагент
            |   И ДоговорыКонтрагентов.Наименование ПОДОБНО &Наименование
            |   И НЕ ДоговорыКонтрагентов.ПометкаУдаления
            |УПОРЯДОЧИТЬ ПО
            |   ДоговорыКонтрагентов.Дата ВОЗР";
            
            Запрос.УстановитьПараметр("Контрагент", КонтрагентСсылка);
            Запрос.УстановитьПараметр("Наименование", "%" + СокрЛП(НаименованиеДоговораИзExcel) + "%");
            
            Выборка = Запрос.Выполнить().Выбрать();
            Если Выборка.Следующий() Тогда
                Возврат Выборка.Ссылка;
            КонецЕсли;
        Исключение
            LogException("ПолучитьИлиСоздатьДоговор - поиск по наименованию");
        КонецПопытки;
    КонецЕсли;
    
    // 2. Ищем договор с нужной целью (самый ранний)
    Попытка
        Запрос = Новый Запрос;
        Запрос.Текст =
        "ВЫБРАТЬ ПЕРВЫЕ 1
        |   ДоговорыКонтрагентов.Ссылка КАК Ссылка
        |ИЗ
        |   Справочник.ДоговорыКонтрагентов КАК ДоговорыКонтрагентов
        |ГДЕ
        |   ДоговорыКонтрагентов.Контрагент = &Контрагент
        |   И ДоговорыКонтрагентов.ТипДоговора = &ТипДоговора
        |   И НЕ ДоговорыКонтрагентов.ПометкаУдаления
        |УПОРЯДОЧИТЬ ПО
        |   ДоговорыКонтрагентов.Дата ВОЗР";
        
        Запрос.УстановитьПараметр("Контрагент", КонтрагентСсылка);
        Запрос.УстановитьПараметр("ТипДоговора", ЦельДоговора);
        
        Выборка = Запрос.Выполнить().Выбрать();
        Если Выборка.Следующий() Тогда
            Возврат Выборка.Ссылка;
        КонецЕсли;
    Исключение
        LogException("ПолучитьИлиСоздатьДоговор - поиск по цели");
    КонецПопытки;
    
    // 3. Ищем любой неудалённый договор контрагента (самый ранний)
    Попытка
        Запрос = Новый Запрос;
        Запрос.Текст =
        "ВЫБРАТЬ ПЕРВЫЕ 1
        |   ДоговорыКонтрагентов.Ссылка КАК Ссылка
        |ИЗ
        |   Справочник.ДоговорыКонтрагентов КАК ДоговорыКонтрагентов
        |ГДЕ
        |   ДоговорыКонтрагентов.Контрагент = &Контрагент
        |   И НЕ ДоговорыКонтрагентов.ПометкаУдаления
        |УПОРЯДОЧИТЬ ПО
        |   ДоговорыКонтрагентов.Дата ВОЗР";
        
        Запрос.УстановитьПараметр("Контрагент", КонтрагентСсылка);
        
        Выборка = Запрос.Выполнить().Выбрать();
        Если Выборка.Следующий() Тогда
            Возврат Выборка.Ссылка;
        КонецЕсли;
    Исключение
        LogException("ПолучитьИлиСоздатьДоговор - любой договор");
    КонецПопытки;
    
    // 4. Договоров нет — создаём новый
    Попытка
        НовыйДоговор = Справочники.ДоговорыКонтрагентов.СоздатьЭлемент();
        
        // Определяем партнёра контрагента
        ПартнерКонтрагента = Неопределено;
        Попытка
            ПартнерКонтрагента = КонтрагентСсылка.Партнер;
        Исключение
        КонецПопытки;
        
        Если ЭтоПоступление Тогда
            НовыйДоговор.Наименование = "Основной договор реализации (авто)";
        Иначе
            НовыйДоговор.Наименование = "Основной договор закупки (авто)";
        КонецЕсли;
        
        НовыйДоговор.Контрагент          = КонтрагентСсылка;
        НовыйДоговор.Организация          = ОрганизацияСсылка;
        НовыйДоговор.ТипДоговора          = ЦельДоговора;
        НовыйДоговор.Статус               = Перечисления.СтатусыДоговоровКонтрагентов.Действует;
        НовыйДоговор.Дата                 = ТекущаяДата();
        НовыйДоговор.ДатаНачалаДействия   = НачалоГода(ТекущаяДата());
        НовыйДоговор.ПорядокРасчетов      = Перечисления.ДетализацияРасчетов.ПоРасчетнымДокументам;
        НовыйДоговор.НалогообложениеНДС   = Перечисления.ТипыНалогообложенияНДС.ПродажаОблагаетсяНДС;
        
        // Валюта взаиморасчетов — рубли
        Попытка
            НовыйДоговор.ВалютаВзаиморасчетов = Справочники.Валюты.НайтиПоКоду("643");
            Если НЕ ЗначениеЗаполнено(НовыйДоговор.ВалютаВзаиморасчетов) Тогда
                НовыйДоговор.ВалютаВзаиморасчетов = Справочники.Валюты.НайтиПоНаименованию("руб.");
            КонецЕсли;
        Исключение
        КонецПопытки;
        
        // Партнёр
        Если ЗначениеЗаполнено(ПартнерКонтрагента) Тогда
            НовыйДоговор.Партнер = ПартнерКонтрагента;
        КонецЕсли;
        
        НовыйДоговор.Записать();
        
        LogLine("Создан новый договор: " + НовыйДоговор.Наименование
            + " для контрагента " + Строка(КонтрагентСсылка));
        
        Возврат НовыйДоговор.Ссылка;
    Исключение
        LogException("ПолучитьИлиСоздатьДоговор - создание нового");
        Возврат Справочники.ДоговорыКонтрагентов.ПустаяСсылка();
    КонецПопытки;

// Функция определяет валюту документа по текстовому наименованию.
// По умолчанию — российский рубль (RUB).
//
// Параметры:
//   ТекстВалюта - Строка - текстовое наименование валюты из Excel
//
// Возвращаемое значение:
//   СправочникСсылка.Валюты - найденная валюта
//
&НаСервереБезКонтекста
Функция ОпределитьВалюту(ТекстВалюта)
    
    Текст = ВРег(СокрЛП(ТекстВалюта));
    
    // По умолчанию — рубль
    Если ПустаяСтрока(Текст)
        Или Текст = "РУБ."
        Или Текст = "РУБ"
        Или Текст = "RUB"
        Или Текст = "RUR"
        Или Текст = "643" Тогда
        
        Результат = Справочники.Валюты.НайтиПоКоду("643");
        Если НЕ ЗначениеЗаполнено(Результат) Тогда
            Результат = Справочники.Валюты.НайтиПоНаименованию("руб.");
        КонецЕсли;
        Возврат Результат;
    КонецЕсли;
    
    // Попытка найти по наименованию
    Попытка
        Результат = Справочники.Валюты.НайтиПоНаименованию(СокрЛП(ТекстВалюта));
        Если ЗначениеЗаполнено(Результат) Тогда
            Возврат Результат;
        КонецЕсли;
    Исключение
    КонецПопытки;
    
    // Если ничего не нашли — пробуем по коду
    Попытка
        Результат = Справочники.Валюты.НайтиПоКоду(СокрЛП(ТекстВалюта));
        Если ЗначениеЗаполнено(Результат) Тогда
            Возврат Результат;
        КонецЕсли;
    Исключение
    КонецПопытки;
    
    // Возвращаем рубль по умолчанию
    Возврат Справочники.Валюты.НайтиПоКоду("643");
    
КонецФункции

        НаборЗаписей = РегистрыСведений.КАА_СвязиДокументовБП.СоздатьНаборЗаписей();
        НаборЗаписей.Отбор.ИдентификаторБП.Установить(ИдентификаторБП);
        
        НоваяЗапись = НаборЗаписей.Добавить();
        НоваяЗапись.ИдентификаторБП = ИдентификаторБП;
        НоваяЗапись.ДокументУТ      = ДокументУТ;
        НоваяЗапись.ДатаЗагрузки    = ТекущаяДата();
        НоваяЗапись.ИмяФайла        = ИмяФайла;
        
        НаборЗаписей.Записать(Истина);
    Исключение
        LogException("ЗаписатьСвязьДокумента: " + ИдентификаторБП);
    КонецПопытки;
    
КонецПроцедуры

// Функция пытается создать банковский счёт контрагента, если он не найден.
//
// Параметры:
//   КонтрагентСсылка - СправочникСсылка.Контрагенты - контрагент
//   НомерСчета       - Строка - номер расчётного счёта
//   БИК              - Строка - БИК банка контрагента

//   СправочникСсылка.БанковскиеСчетаКонтрагентов - созданный счёт, или пустая ссылка
//
&НаСервереБезКонтекста
Функция ПопытатьсяСоздатьСчетКонтрагента(КонтрагентСсылка, НомерСчета, БИК)
    
    Если НЕ ЗначениеЗаполнено(КонтрагентСсылка) Или ПустаяСтрока(НомерСчета) Тогда
        Возврат Справочники.БанковскиеСчетаКонтрагентов.ПустаяСсылка();
    КонецЕсли;
    
    Попытка
        НовыйСчет = Справочники.БанковскиеСчетаКонтрагентов.СоздатьЭлемент();

        НовыйСчет.НомерСчета = НомерСчета;

        // Пытаемся найти банк по БИК
        Если НЕ ПустаяСтрока(БИК) Тогда
            Попытка
                БанкСсылка = Справочники.КлассификаторБанков.НайтиПоКоду(БИК);
                Если ЗначениеЗаполнено(БанкСсылка) Тогда
                    НовыйСчет.Банк = БанкСсылка;
                КонецЕсли;
            Исключение
            КонецПопытки;
        КонецЕсли;
        
        НовыйСчет.Записать();
        
        LogLine("Создан банковский счёт контрагента: " + НомерСчета
            + " для " + Строка(КонтрагентСсылка));
        
        Возврат НовыйСчет.Ссылка;
    Исключение
        LogException("ПопытатьсяСоздатьСчетКонтрагента: " + НомерСчета);
        Возврат Справочники.БанковскиеСчетаКонтрагентов.ПустаяСсылка();
    КонецПопытки;

// ═══════════════════════════════════════════════════════════════════════════
// СЕРВЕРНАЯ ПРОЦЕДУРА: Создание документов в базе 1С
// ═══════════════════════════════════════════════════════════════════════════
//
// Логика работы:
//   1. Обходит ТаблицуДанных формы построчно
//   2. Пропускает строки со статусом "Ошибка" и "Дубль"
//   3. Определяет тип документа: Поступление или Списание (по колонке НаправлениеДенег)
//   4. Создаёт документ, заполняет реквизиты шапки согласно маппингу
//   5. Ищет/создаёт договор контрагента по описанному алгоритму
//   6. Записывает связь в регистр для контроля дублей
//   7. Возвращает результат в виде текстового сообщения
//
&НаСервере
Процедура ЗагрузитьВБазуСервер()
    
    LogLine("=== СТАРТ загрузки документов в базу ===");
    
    // Счётчики результатов
    КоличествоСоздано     = 0;
    КоличествоОшибок      = 0;
    КоличествоПропущено   = 0;
    
    // Список ошибок для вывода пользователю
    СписокОшибок = Новый Массив;
    
    // Имя файла для записи в регистр связей
    ИмяФайлаИсточника = ЭтаФорма.ИмяФайла;
    
    // ─── Основной цикл по строкам таблицы ───
    Для Каждого СтрокаДанных Из ТаблицаДанных Цикл
        
        // Пропускаем строки с ошибками и дубли
        Если СтрокаДанных.Статус = "Ошибка" Тогда
            КоличествоПропущено = КоличествоПропущено + 1;
            Продолжить;
        КонецЕсли;
        
        Если СтрокаДанных.Статус = "Дубль" Тогда
            КоличествоПропущено = КоличествоПропущено + 1;
            Продолжить;
        КонецЕсли;
        
        // Только строки со статусом "Готов"
        Если СтрокаДанных.Статус <> "Готов" Тогда
            КоличествоПропущено = КоличествоПропущено + 1;
            Продолжить;
        КонецЕсли;
        
        // ─── Обработка строки в защищённом блоке ───
        Попытка
        
        // ─── Определение типа документа ───
        НаправлениеВРег = ВРег(СокрЛП(СтрокаДанных.НаправлениеДенег));
        
        ЭтоПоступление = (СтрНайти(НаправлениеВРег, "ПРИХОД") > 0)
            Или (СтрНайти(НаправлениеВРег, "ПОСТУПЛЕНИ") > 0);
            
        ЭтоСписание = (СтрНайти(НаправлениеВРег, "РАСХОД") > 0)
            Или (СтрНайти(НаправлениеВРег, "СПИСАНИ") > 0);
        
        // Если не удалось определить тип — пробуем по колонке ТипДокумента
        Если НЕ ЭтоПоступление И НЕ ЭтоСписание Тогда
            ТипДокВРег = ВРег(СокрЛП(СтрокаДанных.ТипДокумента));
            Если СтрНайти(ТипДокВРег, "ПОСТУПЛЕНИ") > 0 Тогда
                ЭтоПоступление = Истина;
            ИначеЕсли СтрНайти(ТипДокВРег, "СПИСАНИ") > 0 Тогда
                ЭтоСписание = Истина;
            КонецЕсли;
        КонецЕсли;
        
        Если НЕ ЭтоПоступление И НЕ ЭтоСписание Тогда
            КоличествоОшибок = КоличествоОшибок + 1;
            СписокОшибок.Добавить("Строка " + Строка(СтрокаДанных.НомерСтроки)
                + ": не определён тип документа (НаправлениеДенег='"
                + СтрокаДанных.НаправлениеДенег + "')");
            СтрокаДанных.Статус = "Ошибка";
            СтрокаДанных.Комментарий = СтрокаДанных.Комментарий
                + "Не определён тип документа. ";
            Продолжить;
        КонецЕсли;
        
        // ─── Определение хозяйственной операции ───
        ХозОперация = ОпределитьХозяйственнуюОперацию(СтрокаДанных.ОперацияУТ);
        Если ХозОперация = Неопределено Тогда
            // Если операция не найдена — ставим значение по умолчанию
            Если ЭтоПоступление Тогда
                ХозОперация = Перечисления.ХозяйственныеОперации.ПоступлениеОплатыОтКлиента;
            Иначе
                ХозОперация = Перечисления.ХозяйственныеОперации.ОплатаПоставщику;
            КонецЕсли;
            LogLine("Строка " + Строка(СтрокаДанных.НомерСтроки)
                + ": операция '" + СтрокаДанных.ОперацияУТ
                + "' не найдена, подставлено значение по умолчанию");
        КонецЕсли;
        
        // ─── Определение валюты ───
        ВалютаДокумента = ОпределитьВалюту(СтрокаДанных.Валюта);
        
        // ─── Определение договора ───
        ДоговорСсылка = ПолучитьИлиСоздатьДоговор(
            СтрокаДанных.КонтрагентСсылка,
            СтрокаДанных.ОрганизацияСсылка,
            ЭтоПоступление,
            СтрокаДанных.Договор);
        
        // ─── Определение партнёра ───
        ПартнерСсылка = ПолучитьПартнераКонтрагента(СтрокаДанных.КонтрагентСсылка);
        
        // ─── Банковский счёт контрагента ───
        // Если счёт не найден при предпросмотре — пробуем создать
        СчетКонтрагентаСсылка = СтрокаДанных.БанковскийСчетКонтрагента;
        Если НЕ ЗначениеЗаполнено(СчетКонтрагентаСсылка)
            И ЗначениеЗаполнено(СтрокаДанных.КонтрагентСсылка)
            И НЕ ПустаяСтрока(СтрокаДанных.СчетКонтр) Тогда
            
            СчетКонтрагентаСсылка = ПопытатьсяСоздатьСчетКонтрагента(
                СтрокаДанных.КонтрагентСсылка,
                СтрокаДанных.СчетКонтр,
                СтрокаДанных.БИККонтр);
        КонецЕсли;
        
        // Логируем если счёт контрагента пустой (для отладки)
        Если НЕ ЗначениеЗаполнено(СчетКонтрагентаСсылка) Тогда
            LogLine("Строка " + Строка(СтрокаДанных.НомерСтроки)
                + ": Банковский счёт контрагента не заполнен"
                + " (операция: " + СтрокаДанных.ОперацияУТ + ")."
                + " Поле БанковскийСчетКонтрагента не будет установлено.");
        КонецЕсли;
        
        // ─── Этап: Создание документа в транзакции ───
        ЭтапОбработки = "Создание документа";
        Попытка
            НачатьТранзакцию();
            
            Если ЭтоПоступление Тогда
                // ═══════════════════════════════════════════
                // ПОСТУПЛЕНИЕ БЕЗНАЛИЧНЫХ ДЕНЕЖНЫХ СРЕДСТВ
                // ═══════════════════════════════════════════
                ЭтапОбработки = "Создание документа ПоступлениеБезналичныхДенежныхСредств";
                ДокОбъект = Документы.ПоступлениеБезналичныхДенежныхСредств.СоздатьДокумент();
                
                // --- Основные реквизиты ---
                ЭтапОбработки = "Заполнение основных реквизитов (Поступление)";
                ДокОбъект.Дата                        = СтрокаДанных.Дата;
                ДокОбъект.Организация                 = СтрокаДанных.ОрганизацияСсылка;
                ДокОбъект.БанковскийСчет              = СтрокаДанных.БанковскийСчетОрганизации;
                ДокОбъект.ХозяйственнаяОперация       = ХозОперация;
                ДокОбъект.СуммаДокумента              = СтрокаДанных.Сумма;
                ДокОбъект.Контрагент                  = СтрокаДанных.КонтрагентСсылка;
                ДокОбъект.Валюта                       = ВалютаДокумента;
                
                // --- Тип платежного документа (всегда "Платежное поручение") ---
                ДокОбъект.ТипПлатежногоДокумента      = ПолучитьТипПлатежногоДокумента();
                
                // --- Реквизиты входящего документа ---
                ДокОбъект.ДатаВходящегоДокумента      = СтрокаДанных.Дата;
                ДокОбъект.НомерВходящегоДокумента     = СтрокаДанных.НомерПП;
                
                // --- Назначение платежа ---
                ДокОбъект.НазначениеПлатежа            = СтрокаДанных.Назначение;
                
                // --- Комментарий (всегда "Автообмен") ---
                ДокОбъект.Комментарий                  = "Автообмен";
                
                // --- Статья ДДС ---
                ЭтапОбработки = "Установка статьи ДДС (Поступление)";
                Если ЗначениеЗаполнено(СтрокаДанных.СтатьяДДССсылка) Тогда
                    ДокОбъект.СтатьяДвиженияДенежныхСредств = СтрокаДанных.СтатьяДДССсылка;
                КонецЕсли;
                
                // --- Банковский счёт контрагента (ТОЛЬКО если заполнен) ---
                ЭтапОбработки = "Установка банковского счёта контрагента (Поступление)";
                Если ЗначениеЗаполнено(СчетКонтрагентаСсылка) Тогда
                    ДокОбъект.БанковскийСчетКонтрагента = СчетКонтрагентаСсылка;
                КонецЕсли;
                
                // --- Проведено банком ---
                ДокОбъект.ПроведеноБанком              = Истина;
                ДокОбъект.ДатаПроведенияБанком         = СтрокаДанных.Дата;
                
                // --- Договор ---
                Если ЗначениеЗаполнено(ДоговорСсылка) Тогда
                    ДокОбъект.Договор = ДоговорСсылка;
                КонецЕсли;
                
                // --- Партнёр ---
                Если ЗначениеЗаполнено(ПартнерСсылка) Тогда
                    ДокОбъект.Партнер = ПартнерСсылка;
                КонецЕсли;
                
                // --- Имя контрагента (текстовое из выписки) ---
                ДокОбъект.ИмяКонтрагента              = СтрокаДанных.Контрагент;
                
            Иначе
                // ═══════════════════════════════════════════
                // СПИСАНИЕ БЕЗНАЛИЧНЫХ ДЕНЕЖНЫХ СРЕДСТВ
                // ═══════════════════════════════════════════
                ЭтапОбработки = "Создание документа СписаниеБезналичныхДенежныхСредств";
                ДокОбъект = Документы.СписаниеБезналичныхДенежныхСредств.СоздатьДокумент();
                
                // --- Основные реквизиты ---
                ЭтапОбработки = "Заполнение основных реквизитов (Списание)";
                ДокОбъект.Дата                        = СтрокаДанных.Дата;
                ДокОбъект.Организация                 = СтрокаДанных.ОрганизацияСсылка;
                ДокОбъект.БанковскийСчет              = СтрокаДанных.БанковскийСчетОрганизации;
                ДокОбъект.ХозяйственнаяОперация       = ХозОперация;
                ДокОбъект.СуммаДокумента              = СтрокаДанных.Сумма;
                ДокОбъект.Контрагент                  = СтрокаДанных.КонтрагентСсылка;
                ДокОбъект.Валюта                       = ВалютаДокумента;
                
                // --- Тип платежного документа (всегда "Платежное поручение") ---
                ДокОбъект.ТипПлатежногоДокумента      = ПолучитьТипПлатежногоДокумента();
                
                // --- Реквизиты входящего документа ---
                ДокОбъект.ДатаВходящегоДокумента      = СтрокаДанных.Дата;
                ДокОбъект.НомерВходящегоДокумента     = СтрокаДанных.НомерПП;
                
                // --- Назначение платежа ---
                ДокОбъект.НазначениеПлатежа            = СтрокаДанных.Назначение;
                
                // --- Комментарий (всегда "Автообмен") ---
                ДокОбъект.Комментарий                  = "Автообмен";
                
                // --- Статья ДДС ---
                ЭтапОбработки = "Установка статьи ДДС (Списание)";
                Если ЗначениеЗаполнено(СтрокаДанных.СтатьяДДССсылка) Тогда
                    ДокОбъект.СтатьяДвиженияДенежныхСредств = СтрокаДанных.СтатьяДДССсылка;
                КонецЕсли;
                
                // --- Банковский счёт контрагента (ТОЛЬКО если заполнен) ---
                ЭтапОбработки = "Установка банковского счёта контрагента (Списание)";
                Если ЗначениеЗаполнено(СчетКонтрагентаСсылка) Тогда
                    ДокОбъект.БанковскийСчетКонтрагента = СчетКонтрагентаСсылка;
                КонецЕсли;
                
                // --- Проведено банком ---
                ДокОбъект.ПроведеноБанком              = Истина;
                ДокОбъект.ДатаПроведенияБанком         = СтрокаДанных.Дата;
                
                // --- Договор ---
                Если ЗначениеЗаполнено(ДоговорСсылка) Тогда
                    ДокОбъект.Договор = ДоговорСсылка;
                КонецЕсли;
                
                // --- Партнёр ---
                Если ЗначениеЗаполнено(ПартнерСсылка) Тогда
                    ДокОбъект.Партнер = ПартнерСсылка;
                КонецЕсли;
                
                // --- Имя контрагента (текстовое из выписки) ---
                ДокОбъект.ИмяКонтрагента              = СтрокаДанных.Контрагент;
                
            КонецЕсли;
            
            // ─── Запись и проведение документа ───
            ЭтапОбработки = "Запись и проведение документа";
            ДокОбъект.Записать(РежимЗаписиДокумента.Проведение);
            
            // ─── ПРОВЕРКА: Документ реально записался ───
            Если ДокОбъект.Ссылка.Пустая() Тогда
                ВызватьИсключение "Документ не был создан: ссылка пустая после записи";
            КонецЕсли;
            
            // ─── ПРОВЕРКА: Документ реально проведён ───
            Если НЕ ДокОбъект.Проведен Тогда
                ВызватьИсключение "Документ записан, но НЕ проведён. Ссылка: " 
                    + Строка(ДокОбъект.Ссылка);
            КонецЕсли;
            
            ЗафиксироватьТранзакцию();
            
            // ─── Регистрация связи ТОЛЬКО после успешной записи и проведения ───
            // ВАЖНО: Запись в регистр выполняется ПОСЛЕ фиксации транзакции документа,
            // чтобы гарантировать что документ реально существует в базе
            ЭтапОбработки = "Запись связи в регистр сведений";
            ЗаписатьСвязьДокумента(ДокОбъект.Ссылка, СтрокаДанных.УИД, ИмяФайлаИсточника);
            
            // Обновляем статус в таблице формы
            СтрокаДанных.Статус = "Загружено";
            СтрокаДанных.Комментарий = СтрокаДанных.Комментарий
                + "Документ " + Строка(ДокОбъект.Ссылка) + " создан и проведён. ";
            
            КоличествоСоздано = КоличествоСоздано + 1;
            
            // Лог первых 10 созданных документов
            Если КоличествоСоздано <= 10 Тогда
                LogLine("Создан и проведён: " + ?(ЭтоПоступление, "Поступление", "Списание")
                    + " | " + Формат(СтрокаДанных.Дата, "ДЛФ=DD")
                    + " | " + Строка(СтрокаДанных.Контрагент)
                    + " | " + Формат(СтрокаДанных.Сумма, "ЧДЦ=2")
                    + " | Ссылка: " + Строка(ДокОбъект.Ссылка));
            КонецЕсли;
            
        Исключение
            // Откатываем транзакцию если она активна
            Попытка
                ОтменитьТранзакцию();
            Исключение
                // Транзакция могла быть уже зафиксирована или не начата
            КонецПопытки;
            
            ТекстОшибки = ОписаниеОшибки();
            КоличествоОшибок = КоличествоОшибок + 1;
            
            СтрокаДанных.Статус = "Ошибка";
            СтрокаДанных.Комментарий = СтрокаДанных.Комментарий
                + "Ошибка на этапе [" + ЭтапОбработки + "]: " + ТекстОшибки + " ";
            
            СписокОшибок.Добавить("Строка " + Строка(СтрокаДанных.НомерСтроки)
                + " (" + СтрокаДанных.Контрагент + ") [" + ЭтапОбработки + "]: " + ТекстОшибки);
            
            LogException("ЗагрузитьВБазу - строка " + Строка(СтрокаДанных.НомерСтроки)
                + " - этап: " + ЭтапОбработки);
        КонецПопытки;
        
        // ─── Закрытие внешнего защищённого блока обработки строки ───
        Исключение
            ТекстОшибкиСтроки = ОписаниеОшибки();
            КоличествоОшибок = КоличествоОшибок + 1;
            
            СтрокаДанных.Статус = "Ошибка";
            СтрокаДанных.Комментарий = СтрокаДанных.Комментарий
                + "Непредвиденная ошибка на этапе подготовки реквизитов: " + ТекстОшибкиСтроки + " ";
            
            СписокОшибок.Добавить("Строка " + Строка(СтрокаДанных.НомерСтроки)
                + " (подготовка реквизитов): " + ТекстОшибкиСтроки);
            
            LogException("ЗагрузитьВБазу - подготовка реквизитов строки " + Строка(СтрокаДанных.НомерСтроки));
        КонецПопытки;
        
    КонецЦикла;
    
    // ─── Итоговое сообщение ───
    ТекстИтога = "Загрузка завершена."
        + Символы.ПС + "Создано документов: " + Строка(КоличествоСоздано)
        + Символы.ПС + "Пропущено строк: " + Строка(КоличествоПропущено)
        + Символы.ПС + "Ошибок: " + Строка(КоличествоОшибок);
    
    LogLine("=== ИТОГ загрузки: Создано=" + Строка(КоличествоСоздано)
        + " Пропущено=" + Строка(КоличествоПропущено)
        + " Ошибок=" + Строка(КоличествоОшибок) + " ===");
    
    // Выводим итог
    Если КоличествоОшибок = 0 Тогда
        Сообщить(ТекстИтога, СтатусСообщения.Информация);
    Иначе
        Сообщить(ТекстИтога, СтатусСообщения.Важное);
        
        // Выводим первые 20 ошибок
        МаксОшибок = Мин(СписокОшибок.Количество(), 20);
        Для Инд = 0 По МаксОшибок - 1 Цикл
            Сообщить("  ⚠ " + СписокОшибок[Инд], СтатусСообщения.Важное);
        КонецЦикла;
        
        Если СписокОшибок.Количество() > 20 Тогда
            Сообщить("  ... и ещё "
                + Строка(СписокОшибок.Количество() - 20) + " ошибок",
                СтатусСообщения.Важное);
        КонецЕсли;
    КонецЕсли;
    
КонецПроцедуры

// КЛИЕНТСКАЯ ПРОЦЕДУРА: Обработчик нажатия кнопки "Загрузить в базу"
// ═══════════════════════════════════════════════════════════════════════════
//
// Кнопка вызывает эту процедуру. Она проверяет наличие данных,
// запрашивает подтверждение у пользователя и вызывает серверную загрузку.
//
&НаКлиенте
Процедура ЗагрузитьВБазу(Команда)
    
    // Проверяем, что таблица заполнена
    Если ТаблицаДанных.Количество() = 0 Тогда
        ПоказатьПредупреждение(, "Таблица данных пуста. Сначала загрузите файл Excel.");
        Возврат;
    КонецЕсли;
    
    // Подсчитываем количество строк к загрузке
    КоличествоГотовых = 0;
    КоличествоДублей  = 0;
    КоличествоОшибок  = 0;
    
    Для Каждого СтрокаТаблицы Из ТаблицаДанных Цикл
        Если СтрокаТаблицы.Статус = "Готов" Тогда
            КоличествоГотовых = КоличествоГотовых + 1;
        ИначеЕсли СтрокаТаблицы.Статус = "Дубль" Тогда
            КоличествоДублей = КоличествоДублей + 1;
        ИначеЕсли СтрокаТаблицы.Статус = "Ошибка" Тогда
            КоличествоОшибок = КоличествоОшибок + 1;
        КонецЕсли;
    КонецЦикла;
    
    Если КоличествоГотовых = 0 Тогда
        ПоказатьПредупреждение(, "Нет строк со статусом ""Готов"" для загрузки."
            + Символы.ПС + "Дублей: " + Строка(КоличествоДублей)
            + Символы.ПС + "Ошибок: " + Строка(КоличествоОшибок));
        Возврат;
    КонецЕсли;
    
    // Запрашиваем подтверждение
    ТекстВопроса = "Будет создано документов: " + Строка(КоличествоГотовых)
        + Символы.ПС + "Пропущено (дубли): " + Строка(КоличествоДублей)
        + Символы.ПС + "Пропущено (ошибки): " + Строка(КоличествоОшибок)
        + Символы.ПС + Символы.ПС + "Продолжить загрузку?";
    
    ОписаниеОповещения = Новый ОписаниеОповещения(
        "ЗагрузитьВБазуПодтверждение", ЭтотОбъект);
    
    ПоказатьВопрос(ОписаниеОповещения, ТекстВопроса, РежимДиалогаВопрос.ДаНет, ,
        КодВозвратаДиалога.Нет, "Загрузка в базу");
    
КонецПроцедуры
