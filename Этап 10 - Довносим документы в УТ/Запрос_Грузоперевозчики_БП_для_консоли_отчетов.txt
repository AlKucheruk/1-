// 1С:БП 3.0 — суммы по поставщикам за услуги грузоперевозки / автотранспорта
// Под: Управляемая консоль отчетов 3.8.9. Период зашит: 01.01.2016 — 01.01.2027 (вкл. конец суток).
//
// Метаданные сверены с выгрузкой D:\1СБП:
//   Документ.ПоступлениеТоваровУслуг (шапка: Контрагент; ТЧ: Услуги, Товары)
//   Справочник.Номенклатура: Услуга, Наименование
//
// Логика: по проведённым "Поступление (акты, накладные, УПД)" суммируем строки ТЧ
//   «Услуги» и «Товары», удовлетворяющие шаблону «перевозка» (см. условие ниже).
// Сумма по строке: Сумма + СуммаНДС (всего с НДС). При необходимости оставьте
//   только Сумма (без НДС) — замените выражение в Влож.СуммаСтроки.
//
// ВАЖНО: подставьте свои слова в ПОДОБНО или укажите папку номенклатуры в ИЕРАРХИИ.
// Если часть расходов вводится "Поступлением доп. расходов" — добавьте ОБЪЕДИНИТЬ
//   по аналогии с этим же отбором по номенклатуре/содержанию.
// =============================================================================

ВЫБРАТЬ
	Влож.Контрагент КАК Контрагент,
	Влож.ИНН КАК ИНН,
	СУММА(Влож.СуммаСтроки) КАК СуммаВсегоСНДС
ИЗ
	(
		ВЫБРАТЬ
			Поступление.Контрагент КАК Контрагент,
			ЕСТЬNULL(Поступление.Контрагент.ИНН, "") КАК ИНН,
			ВЫРАЗИТЬ(ЕСТЬNULL(Дет.Сумма, 0) + ЕСТЬNULL(Дет.СуммаНДС, 0) КАК ЧИСЛО(15, 2)) КАК СуммаСтроки
		ИЗ
			Документ.ПоступлениеТоваровУслуг.Услуги КАК Дет
				ВНУТРЕННЕЕ СОЕДИНЕНИЕ Документ.ПоступлениеТоваровУслуг КАК Поступление
				ПО Дет.Ссылка = Поступление.Ссылка
		ГДЕ
			Поступление.Проведен
			И НЕ Поступление.ПометкаУдаления
			И Поступление.Дата >= ДАТАВРЕМЯ(2016, 1, 1, 0, 0, 0)
			И Поступление.Дата <= КОНЕЦПЕРИОДА(ДАТАВРЕМЯ(2027, 1, 1, 0, 0, 0), ДЕНЬ)
			// --- отбор "перевозки": расширьте/измените под свою номенклатуру
			И (
				ВЫРАЗИТЬ(Дет.Содержание КАК СТРОКА(1000)) ПОДОБНО "%перевоз%"
				ИЛИ ВЫРАЗИТЬ(Дет.Содержание КАК СТРОКА(1000)) ПОДОБНО "%грузоперевоз%"
				ИЛИ ВЫРАЗИТЬ(Дет.Содержание КАК СТРОКА(1000)) ПОДОБНО "%доставк%"
				ИЛИ ВЫРАЗИТЬ(Дет.Содержание КАК СТРОКА(1000)) ПОДОБНО "%автотранспорт%"
				ИЛИ ВЫРАЗИТЬ(Дет.Содержание КАК СТРОКА(1000)) ПОДОБНО "%транспорт%"
				ИЛИ ВЫРАЗИТЬ(Дет.Номенклатура.Наименование КАК СТРОКА(200)) ПОДОБНО "%перевоз%"
				ИЛИ ВЫРАЗИТЬ(Дет.Номенклатура.Наименование КАК СТРОКА(200)) ПОДОБНО "%грузоперевоз%"
				ИЛИ ВЫРАЗИТЬ(Дет.Номенклатура.НаименованиеПолное КАК СТРОКА(1000)) ПОДОБНО "%перевоз%"
				// раскомментируйте, если в каталоге есть папка перевозчиков:
				// ИЛИ Дет.Номенклатура В ИЕРАРХИИ (&СсылкаПапкиНоменклатурыПеревозки)
			)

		ОБЪЕДИНИТЬ ВСЕ

		ВЫБРАТЬ
			Поступление.Контрагент,
			ЕСТЬNULL(Поступление.Контрагент.ИНН, ""),
			ВЫРАЗИТЬ(ЕСТЬNULL(Дет.Сумма, 0) + ЕСТЬNULL(Дет.СуммаНДС, 0) КАК ЧИСЛО(15, 2))
		ИЗ
			Документ.ПоступлениеТоваровУслуг.Товары КАК Дет
				ВНУТРЕННЕЕ СОЕДИНЕНИЕ Документ.ПоступлениеТоваровУслуг КАК Поступление
				ПО Дет.Ссылка = Поступление.Ссылка
		ГДЕ
			Поступление.Проведен
			И НЕ Поступление.ПометкаУдаления
			И Поступление.Дата >= ДАТАВРЕМЯ(2016, 1, 1, 0, 0, 0)
			И Поступление.Дата <= КОНЕЦПЕРИОДА(ДАТАВРЕМЯ(2027, 1, 1, 0, 0, 0), ДЕНЬ)
			// в ТЧ «Товары» нет поля «Содержание» — ориентируемся на номенклатуру
			И Дет.Номенклатура.Услуга
			И (
				ВЫРАЗИТЬ(Дет.Номенклатура.Наименование КАК СТРОКА(200)) ПОДОБНО "%перевоз%"
				ИЛИ ВЫРАЗИТЬ(Дет.Номенклатура.Наименование КАК СТРОКА(200)) ПОДОБНО "%грузоперевоз%"
				ИЛИ ВЫРАЗИТЬ(Дет.Номенклатура.Наименование КАК СТРОКА(200)) ПОДОБНО "%доставк%"
				ИЛИ ВЫРАЗИТЬ(Дет.Номенклатура.Наименование КАК СТРОКА(200)) ПОДОБНО "%автотранспорт%"
				ИЛИ ВЫРАЗИТЬ(Дет.Номенклатура.Наименование КАК СТРОКА(200)) ПОДОБНО "%транспорт%"
				ИЛИ ВЫРАЗИТЬ(Дет.Номенклатура.НаименованиеПолное КАК СТРОКА(1000)) ПОДОБНО "%перевоз%"
			)
	) КАК Влож

	Влож.ИНН
